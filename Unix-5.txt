
 UNIX COMMANDS – EXAMPLES
1️⃣ ls

bash
ls              list files in current folder
ls -l           list files with details (permissions, owner, size)


 2️⃣ cp

bash
cp file1.txt backup_file1.txt      copy a file
cp -r project/ project_backup/     copy a folder recursively


 3️⃣ mv

bash
mv oldname.txt newname.txt         rename a file
mv report.txt /backup/reports/     move file to another folder


 4️⃣ sort

bash
sort names.txt                      sort file alphabetically
sort -r scores.txt                  sort file in reverse order


 5️⃣ grep

bash
grep ERROR system.log               search for "ERROR" in log file
grep -i warning server.log          search for "warning" ignoring case


 6️⃣ cat

bash
cat test.txt                        display file content
cat file1.txt file2.txt              display multiple files


 7️⃣ head

bash
head -n 5 log.txt                   display first 5 lines
head log.txt                         default first 10 lines


 8️⃣ tail

bash
tail -n 5 log.txt                   last 5 lines
tail -f system.log                   follow new log entries in real-time


 9️⃣ man

bash
man ls                               view manual for ls
man grep                             view manual for grep


 1️⃣0️⃣ locate

bash
locate test.txt                      find file by name
locate .log                         find all log files


 1️⃣1️⃣ find

bash
find /home -name test.txt            search file in /home
find /var -type f -size +1M          find files larger than 1MB


 1️⃣2️⃣ diff

bash
diff file1.txt file2.txt             compare 2 files
diff -u file1.txt file2.txt          unified diff format


 1️⃣3️⃣ file

bash
file test.txt                        check file type
file image.png                        check type of image file


 1️⃣4️⃣ rm

bash
rm test.txt                          delete file
rm -r old_folder/                    delete folder recursively


 1️⃣5️⃣ mkdir

bash
mkdir new_folder                      create a folder
mkdir -p project/docs                  create nested folders


 1️⃣6️⃣ rmdir

bash
rmdir empty_folder                     remove empty folder


 1️⃣7️⃣ cd

bash
cd /home/sari                          go to home directory
cd ../                                   go to parent folder


 1️⃣8️⃣ pwd

bash
pwd                                     show current folder path


 1️⃣9️⃣ ln (hard link) & ln -s (symbolic link)

bash
ln file1.txt file1_link                 create hard link
ln -s /home/sari/file1.txt link.txt     create symbolic link


 2️⃣0️⃣ gzip / gunzip

bash
gzip file.txt                           compress file
gunzip file.txt.gz                       decompress file


 2️⃣1️⃣ zip / unzip

bash
zip archive.zip file1.txt file2.txt      create zip archive
unzip archive.zip                        extract zip archive


 2️⃣2️⃣ tar

bash
tar -cvf archive.tar folder/            create tar archive
tar -xvf archive.tar                     extract tar archive


 2️⃣3️⃣ touch

bash
touch newfile.txt                         create empty file
touch file1.txt file2.txt                 create multiple files


 2️⃣4️⃣ echo

bash
echo "Hello Volkswagen" > test.txt        write text to file
echo $HOME                                display value of environment variable


 2️⃣5️⃣ who

bash
who                                       list logged-in users


 2️⃣6️⃣ whoami

bash
whoami                                    display current user


 2️⃣7️⃣ ps

bash
ps                                        list current processes
ps -ef                                    detailed list of processes


 2️⃣8️⃣ kill

bash
kill 1234                                 kill process with PID 1234
kill -9 1234                              force kill process


 2️⃣9️⃣ chmod

bash
chmod 755 script.sh                        set permissions: owner=rwx, group=rx, others=rx
chmod u+x run.sh                           give execute permission to owner


 3️⃣0️⃣ chown & chgrp

bash
chown sari file.txt                        change file owner
chgrp developers file.txt                  change file group




 2 USE CASES FOR ALL TOPICS COMBINED

These 2 use cases are realistic Volkswagen-style corporate examples:

 Use Case 1 – Log File Management

 Scenario: Server generates daily logs in `/var/logs/`.
 Commands used:
  `ls, cd, pwd, cat, tail, grep, sort, diff, gzip, tar`
 Flow:

  1. Navigate to log folder → `cd /var/logs/`
  2. Check logs → `ls -l`
  3. Read last entries → `tail -n 20 system.log`
  4. Search for errors → `grep ERROR system.log`
  5. Sort results → `sort errors.log`
  6. Compress old logs → `gzip oldlog.log`
  7. Archive weekly logs → `tar -cvf logs_week1.tar .gz`

---

 Use Case 2 – File Permissions and Collaboration

 Scenario: Multiple developers working on project files
 Commands used:
  `chmod, chown, chgrp, ln, ln -s, mkdir, touch, rm`
 Flow:

  1. Create project folder → `mkdir project`
  2. Create files → `touch index.html style.css`
  3. Set correct permissions → `chmod 644 index.html`
  4. Assign owner/group → `chown sari index.html` / `chgrp developers index.html`
  5. Create symbolic link for shared config → `ln -s /shared/config.cfg config.cfg`
  6. Remove temp files → `rm temp.txt`
